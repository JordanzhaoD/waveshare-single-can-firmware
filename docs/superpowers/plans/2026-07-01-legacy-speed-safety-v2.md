# Legacy Speed & Safety v2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add Legacy Smart Offset v2 and an experimental default-off Abort Guard while keeping Legacy speed writes limited to `0x2F8 / 760`.

**Architecture:** Add two focused pure modules: `dash_legacy_smart_offset.h` computes a clamped `0..33 km/h` Legacy offset, and `dash_abort_guard.h` tracks AP abort states and exposes injection blocking. Existing handlers remain responsible for CAN writes, dashboard APIs persist config and expose diagnostics, and the UI adds small controls without a full redesign.

**Tech Stack:** ESP-IDF/PlatformIO C++17, ArduinoJson, native PlatformIO Unity tests, Python unittest contract tests, embedded dashboard HTML generated by `scripts/minify_dashboard.py`.

---

## Scope Check

The approved spec covers two tightly related subsystems:

1. Legacy speed offset calculation and `0x2F8` diagnostics.
2. Abort Guard safety gating for injection paths.

They share `0x399` observations, dashboard configuration, `/status` diagnostics, and handler gates. They can be implemented in one plan because each task remains independently testable and the final feature is one coherent dashboard/user-facing capability.

```cpp
#pragma once

#include <cstdint>
#include <algorithm>

inline constexpr uint8_t kLegacySmartOffsetMaxKph = 33;
inline constexpr uint8_t kLegacySmartPctMax = 63;
inline constexpr uint8_t kLegacySmartSmoothRateDefault = 5;
inline constexpr uint8_t kLegacySmartSmoothRateMin = 1;
inline constexpr uint8_t kLegacySmartSmoothRateMax = 20;

enum class LegacySmartOffsetMode : uint8_t
{
    Off = 0,
    Manual = 1,
    Auto = 2,
    Custom = 3,
};

struct LegacySmartOffsetConfig
{
    LegacySmartOffsetMode mode = LegacySmartOffsetMode::Off;
    uint8_t manualOffsetKph = 0;
    bool smoothDownEnabled = true;
    uint8_t smoothDownRateKphS = kLegacySmartSmoothRateDefault;
    uint8_t customPctLow = 50;
    uint8_t customPctMid = 30;
    uint8_t customPctHigh = 20;
    uint8_t customPctVeryHigh = 10;
};

struct LegacySmartOffsetResult
{
    LegacySmartOffsetMode mode = LegacySmartOffsetMode::Off;
    uint8_t speedLimitRaw = 0;
    uint16_t speedLimitKph = 0;
    uint8_t offsetPct = 0;
    uint16_t absoluteCapKph = 0;
    uint16_t rawTargetKph = 0;
    uint16_t smoothedTargetKph = 0;
    uint8_t outputOffsetKph = 0;
    bool fallbackUsed = false;
    bool smoothingActive = false;
    uint32_t lastUpdateMs = 0;
    const char *blockedReason = "none";
};

inline uint8_t dashClampLegacySmartOffsetKph(int value)
{
    if (value < 0)
        return 0;
    if (value > kLegacySmartOffsetMaxKph)
        return kLegacySmartOffsetMaxKph;
    return static_cast<uint8_t>(value);
}

inline uint8_t dashClampLegacySmartPct(int value)
{
    if (value < 0)
        return 0;
    if (value > kLegacySmartPctMax)
        return kLegacySmartPctMax;
    return static_cast<uint8_t>(value);
}

inline uint8_t dashClampLegacySmartRate(int value)
{
    if (value < kLegacySmartSmoothRateMin)
        return kLegacySmartSmoothRateDefault;
    if (value > kLegacySmartSmoothRateMax)
        return kLegacySmartSmoothRateDefault;
    return static_cast<uint8_t>(value);
}

inline LegacySmartOffsetMode dashClampLegacySmartMode(uint8_t value)
{
    if (value > static_cast<uint8_t>(LegacySmartOffsetMode::Custom))
        return LegacySmartOffsetMode::Off;
    return static_cast<LegacySmartOffsetMode>(value);
}

inline bool dashLegacySmartLimitValid(uint8_t raw)
{
    return raw > 0 && raw < 31;
}

inline uint8_t dashLegacySmartAutoPct(uint16_t limitKph)
{
    if (limitKph <= 35)
        return 63;
    if (limitKph <= 60)
        return 50;
    if (limitKph <= 90)
        return 30;
    if (limitKph <= 110)
        return 20;
    return 10;
}

inline uint16_t dashLegacySmartAbsoluteCap(uint16_t limitKph)
{
    switch (limitKph)
    {
    case 40:
        return 60;
    case 45:
        return 67;
    case 50:
        return 75;
    case 55:
        return 82;
    case 60:
        return 90;
    case 70:
        return 91;
    case 80:
        return 104;
    case 90:
        return 117;
    case 100:
        return 120;
    case 110:
        return 132;
    case 120:
        return 132;
    default:
        if (limitKph <= 35)
            return 60;
        return static_cast<uint16_t>(limitKph + 15);
    }
}

inline uint8_t dashLegacySmartCustomPct(const LegacySmartOffsetConfig &cfg, uint16_t limitKph)
{
    if (limitKph <= 50)
        return dashClampLegacySmartPct(cfg.customPctLow);
    if (limitKph <= 70)
        return dashClampLegacySmartPct(cfg.customPctMid);
    if (limitKph <= 100)
        return dashClampLegacySmartPct(cfg.customPctHigh);
    return dashClampLegacySmartPct(cfg.customPctVeryHigh);
}

class LegacySmartOffsetEngine
{
public:
    LegacySmartOffsetResult compute(const LegacySmartOffsetConfig &cfg, uint8_t speedLimitRaw,
                                    uint32_t nowMs, bool apOrFsdEngaged)
    {
        LegacySmartOffsetResult out;
        out.mode = cfg.mode;
        out.speedLimitRaw = speedLimitRaw;
        out.lastUpdateMs = nowMs;

        if (cfg.mode == LegacySmartOffsetMode::Off)
        {
            resetSmoothing();
            out.blockedReason = "off";
            return out;
        }

        if (cfg.mode == LegacySmartOffsetMode::Manual || !dashLegacySmartLimitValid(speedLimitRaw))
        {
            out.fallbackUsed = cfg.mode != LegacySmartOffsetMode::Manual;
            out.outputOffsetKph = dashClampLegacySmartOffsetKph(cfg.manualOffsetKph);
            out.smoothedTargetKph = out.outputOffsetKph;
            if (!dashLegacySmartLimitValid(speedLimitRaw) && cfg.mode != LegacySmartOffsetMode::Manual)
                out.blockedReason = "speedLimitUnknown";
            else
                out.blockedReason = out.outputOffsetKph == 0 ? "manualZero" : "none";
            resetSmoothing();
            return out;
        }

        out.speedLimitKph = static_cast<uint16_t>(speedLimitRaw) * 5U;
        out.offsetPct = cfg.mode == LegacySmartOffsetMode::Custom
                            ? dashLegacySmartCustomPct(cfg, out.speedLimitKph)
                            : dashLegacySmartAutoPct(out.speedLimitKph);
        out.absoluteCapKph = dashLegacySmartAbsoluteCap(out.speedLimitKph);

        uint32_t numerator = static_cast<uint32_t>(out.speedLimitKph) * (100U + out.offsetPct);
        uint16_t target = static_cast<uint16_t>((numerator + 50U) / 100U);
        if (target > out.absoluteCapKph)
            target = out.absoluteCapKph;
        out.rawTargetKph = target;
        out.smoothedTargetKph = smoothTarget(cfg, target, out.speedLimitKph, nowMs, apOrFsdEngaged, out.smoothingActive);

        int desiredOffset = static_cast<int>(out.smoothedTargetKph) - static_cast<int>(out.speedLimitKph);
        out.outputOffsetKph = dashClampLegacySmartOffsetKph(desiredOffset);
        out.blockedReason = out.outputOffsetKph == 0 ? "zeroOutput" : "none";
        return out;
    }

    void resetSmoothing()
    {
        haveSmoothed_ = false;
        smoothedTargetKph_ = 0;
        lastUpdateMs_ = 0;
    }

private:
    uint16_t smoothTarget(const LegacySmartOffsetConfig &cfg, uint16_t rawTargetKph,
                          uint16_t speedLimitKph, uint32_t nowMs, bool engaged,
                          bool &smoothingActive)
    {
        smoothingActive = false;
        uint16_t maxExpressibleTarget = static_cast<uint16_t>(speedLimitKph + kLegacySmartOffsetMaxKph);
        if (rawTargetKph > maxExpressibleTarget)
            rawTargetKph = maxExpressibleTarget;

        if (!haveSmoothed_ || rawTargetKph >= smoothedTargetKph_ || !cfg.smoothDownEnabled || !engaged)
        {
            haveSmoothed_ = true;
            smoothedTargetKph_ = rawTargetKph;
            lastUpdateMs_ = nowMs;
            return smoothedTargetKph_;
        }

        uint32_t dtMs = nowMs >= lastUpdateMs_ ? (nowMs - lastUpdateMs_) : 0;
        if (dtMs == 0 || dtMs > 10000U)
        {
            smoothedTargetKph_ = rawTargetKph;
            lastUpdateMs_ = nowMs;
            return smoothedTargetKph_;
        }

        uint8_t rate = dashClampLegacySmartRate(cfg.smoothDownRateKphS);
        uint32_t drop = (static_cast<uint32_t>(rate) * dtMs + 500U) / 1000U;
        if (drop == 0)
            drop = 1;

        if (smoothedTargetKph_ > rawTargetKph)
        {
            uint16_t floor = smoothedTargetKph_ > drop ? static_cast<uint16_t>(smoothedTargetKph_ - drop) : rawTargetKph;
            smoothedTargetKph_ = floor < rawTargetKph ? rawTargetKph : floor;
            smoothingActive = smoothedTargetKph_ > rawTargetKph;
        }
        lastUpdateMs_ = nowMs;
        return smoothedTargetKph_;
    }

    bool haveSmoothed_ = false;
    uint16_t smoothedTargetKph_ = 0;
    uint32_t lastUpdateMs_ = 0;
};
```

#pragma once

#include <cstdint>

enum class DashAbortGuardBlockPath : uint8_t
{
    None = 0,
    LegacyFsdMux0,
    LegacyFsdMux1,
    LegacySpeed0x2f8,
    LegacyVisionSlider0x438,
    Nag,
};

struct DashAbortGuardDiag
{
    bool enabled = false;
    bool latched = false;
    uint8_t lastApState = 0;
    uint8_t lastAbortState = 0;
    uint32_t latchedAtMs = 0;
    const char *lastClearReason = "none";
    uint32_t blocks = 0;
    const char *lastBlockedPath = "none";
};

inline const char *dashAbortGuardBlockPathName(DashAbortGuardBlockPath path)
{
    switch (path)
    {
    case DashAbortGuardBlockPath::LegacyFsdMux0:
        return "legacy_fsd_mux0";
    case DashAbortGuardBlockPath::LegacyFsdMux1:
        return "legacy_fsd_mux1";
    case DashAbortGuardBlockPath::LegacySpeed0x2f8:
        return "legacy_speed_0x2f8";
    case DashAbortGuardBlockPath::LegacyVisionSlider0x438:
        return "legacy_vision_slider_0x438";
    case DashAbortGuardBlockPath::Nag:
        return "nag";
    case DashAbortGuardBlockPath::None:
    default:
        return "none";
    }
}

class DashAbortGuard
{
public:
    void setEnabled(bool enabled)
    {
        if (enabled_ == enabled)
            return;
        enabled_ = enabled;
        if (!enabled_)
        {
            latched_ = false;
            lastClearReason_ = "disabled";
        }
    }

    void onApState(uint8_t apState, uint32_t nowMs)
    {
        lastApState_ = apState;
        if (!enabled_)
            return;

        if (apState == 8 || apState == 9)
        {
            if (!latched_)
                latchedAtMs_ = nowMs;
            latched_ = true;
            lastAbortState_ = apState;
            lastClearReason_ = "none";
            return;
        }

        if (latched_ && apState < 2)
        {
            latched_ = false;
            lastClearReason_ = "cleanDisengage";
        }
    }

    bool allowsInjection() const
    {
        return !enabled_ || !latched_;
    }

    void recordBlock(DashAbortGuardBlockPath path)
    {
        if (allowsInjection())
            return;
        ++blocks_;
        lastBlockedPath_ = dashAbortGuardBlockPathName(path);
    }

    DashAbortGuardDiag diag() const
    {
        DashAbortGuardDiag d;
        d.enabled = enabled_;
        d.latched = latched_;
        d.lastApState = lastApState_;
        d.lastAbortState = lastAbortState_;
        d.latchedAtMs = latchedAtMs_;
        d.lastClearReason = lastClearReason_;
        d.blocks = blocks_;
        d.lastBlockedPath = lastBlockedPath_;
        return d;
    }

private:
    bool enabled_ = false;
    bool latched_ = false;
    uint8_t lastApState_ = 0;
    uint8_t lastAbortState_ = 0;
    uint32_t latchedAtMs_ = 0;
    const char *lastClearReason_ = "none";
    uint32_t blocks_ = 0;
    const char *lastBlockedPath_ = "none";
};
